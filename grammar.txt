start = sequence

ws "whitespace" = " "*
quote = '"'

// single step definition
step_char =  [0-9a-zA-Z]
cycle_step = chars:step_char+ ws { return chars.join("");}

// define a sub cycle
sub_cycle =  "[" elements:element+ "]" { return elements;}

// a sequence slice is either a single step or a sub cycle
sequence_slice = cycle_step / sub_cycle

// a sequence, made of slices
sequence = ws quote slice:sequence_slice+ quote { return slice; }
